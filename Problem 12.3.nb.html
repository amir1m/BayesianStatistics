count <-0
lsamples <- vector(length = n)
while (count < n){
  x <- rtruncnorm(n = 1, a = 1, b = Inf)
  lsamples[count] <- x^6 * exp(-2 * x)
  count <- count + 1
}

lmean = mean(lsamples)

n=1000
count <- 0
lsamples <- vector(length = n)
while (count < n){
  x <- runif(n = 1, min = 1, max = 10)
  lsamples[count] <- 9 * exp(- x^4 / 2) * x^6 / sqrt(2 * pi)
  count <- count + 1
}

lmean = mean(lsamples)
cat(lmean)
0.2968113
hist(lsamples)
abline(v=lmean)
